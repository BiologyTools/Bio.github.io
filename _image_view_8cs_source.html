419 private void SetLabelMacroResolutions()
420 {
421 int i = 0;
422 AForge.Size s = new AForge.Size(int.MaxValue, int.MaxValue);
423 bool noMacro = true;
424 PixelFormat pf = SelectedImage.Resolutions[0].PixelFormat;
425 foreach (Resolution res in SelectedImage.Resolutions)
426 {
427 if (res.SizeX < s.Width && res.SizeY < s.Height)
428 {
429 //If the pixel format changes it means this is the macro resolution.
430 if (pf != res.PixelFormat)
431 {
432 noMacro = false;
433 break;
434 }
435 pf = res.PixelFormat;
436 s = new AForge.Size(res.SizeX, res.SizeY);
437 }
438 else
439 {
440 //If this level is bigger than the previous this is the macro resolution.
441 noMacro = false;
442 break;
443 }
444 i++;
445 }
446 if (!noMacro)
447 {
448 MacroResolution = i;
449 LabelResolution = i + 1;
450 }
