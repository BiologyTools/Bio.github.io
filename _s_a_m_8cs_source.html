86 public float[] Decode(List<Promotion> promotions, int orgWid, int orgHei)
87 {
88 var embedding_tensor = new DenseTensor<float>(this.mImgEmbedding, new[] { 1, 256, 64, 64 });
89 var bpmos = promotions.FindAll(e => e.mType == PromotionType.Box);
90 var pproms = promotions.FindAll(e => e.mType == PromotionType.Point);
91 int boxCount = promotions.FindAll(e => e.mType == PromotionType.Box).Count();
92 int pointCount = promotions.FindAll(e => e.mType == PromotionType.Point).Count();
93 float[] promotion = new float[2 * (boxCount * 2 + pointCount)];
94 float[] label = new float[boxCount * 2 + pointCount];
95 for (int i = 0; i < boxCount; i++)
96 {
97 var input = bpmos[i].GetInput();
98 for (int j = 0; j < input.Count(); j++)
99 {
100 promotion[4 * i + j] = input[j];
101 }
102 var la = bpmos[i].GetLable();
103 for (int j = 0; j < la.Count(); j++)
104 {
105 label[2 * i + j] = la[j];
106 }
107 }
108 for (int i = 0; i < pointCount; i++)
109 {
110 var p = pproms[i].GetInput();
111 for (int j = 0; j < p.Count(); j++)
112 {
113 promotion[boxCount * 4 + 2 * i + j] = p[j];
114 }
115 var la = pproms[i].GetLable();
116 for (int j = 0; j < la.Count(); j++)
117 {
118 label[boxCount * 2 + i + j] = la[j];
119 }
120 }
121
122 var point_coords_tensor = new DenseTensor<float>(promotion, new[] { 1, boxCount * 2 + pointCount, 2 });
123
124 var point_label_tensor = new DenseTensor<float>(label, new[] { 1, boxCount * 2 + pointCount });
125
126 float[] mask = new float[256 * 256];
127 for (int i = 0; i < mask.Count(); i++)
128 {
129 mask[i] = 0;
130 }
131 var mask_tensor = new DenseTensor<float>(mask, new[] { 1, 1, 256, 256 });
132
133 float[] hasMaskValues = new float[1] { 0 };
134 var hasMaskValues_tensor = new DenseTensor<float>(hasMaskValues, new[] { 1 });
135
136 float[] orig_im_size_values = { (float)orgHei, (float)orgWid };
137 var orig_im_size_values_tensor = new DenseTensor<float>(orig_im_size_values, new[] { 2 });
138
139 var decode_inputs = new List<NamedOnnxValue>
140 {
141 NamedOnnxValue.CreateFromTensor("image_embeddings", embedding_tensor),
142 NamedOnnxValue.CreateFromTensor("point_coords", point_coords_tensor),
143 NamedOnnxValue.CreateFromTensor("point_labels", point_label_tensor),
144 NamedOnnxValue.CreateFromTensor("mask_input", mask_tensor),
145 NamedOnnxValue.CreateFromTensor("has_mask_input", hasMaskValues_tensor),
146 NamedOnnxValue.CreateFromTensor("orig_im_size", orig_im_size_values_tensor)
147 };
148
149 var segmask = this.mDecoder.Run(decode_inputs);
150 var outputmask = segmask.First().AsTensor<float>().ToArray();
151 return outputmask;
