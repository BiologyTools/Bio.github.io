112 public float[] Decode(BioImage b, List<Promotion> promotions, int orgWid, int orgHei)
113 {
114 ZCT c = App.viewer.GetCoordinate();
115 List<float[]> lts = (List<float[]>)b.Tag;
116 var embedding_tensor = new DenseTensor<float>(lts[b.Coords[c.Z,c.C,c.T]], new[] { 1, 256, 64, 64 });
117 var bpmos = promotions.FindAll(e => e.mType == PromotionType.Box);
118 var pproms = promotions.FindAll(e => e.mType == PromotionType.Point);
119 int boxCount = promotions.FindAll(e => e.mType == PromotionType.Box).Count();
120 int pointCount = promotions.FindAll(e => e.mType == PromotionType.Point).Count();
121 float[] promotion = new float[2 * (boxCount * 2 + pointCount)];
122 float[] label = new float[boxCount * 2 + pointCount];
123 for (int i = 0; i < boxCount; i++)
124 {
125 var input = bpmos[i].GetInput();
126 for (int j = 0; j < input.Count(); j++)
127 {
128 promotion[4 * i + j] = input[j];
129 }
130 var la = bpmos[i].GetLable();
131 for (int j = 0; j < la.Count(); j++)
132 {
133 label[2 * i + j] = la[j];
134 }
135 }
136 for (int i = 0; i < pointCount; i++)
137 {
138 var p = pproms[i].GetInput();
139 for (int j = 0; j < p.Count(); j++)
140 {
141 promotion[boxCount * 4 + 2 * i + j] = p[j];
142 }
143 var la = pproms[i].GetLable();
144 for (int j = 0; j < la.Count(); j++)
145 {
146 label[boxCount * 2 + i + j] = la[j];
147 }
148 }
149
150 var point_coords_tensor = new DenseTensor<float>(promotion, new[] { 1, boxCount * 2 + pointCount, 2 });
151
152 var point_label_tensor = new DenseTensor<float>(label, new[] { 1, boxCount * 2 + pointCount });
153
154 float[] mask = new float[256 * 256];
155 for (int i = 0; i < mask.Count(); i++)
156 {
157 mask[i] = 0;
158 }
159 var mask_tensor = new DenseTensor<float>(mask, new[] { 1, 1, 256, 256 });
160
161 float[] hasMaskValues = new float[1] { 0 };
162 var hasMaskValues_tensor = new DenseTensor<float>(hasMaskValues, new[] { 1 });
163
164 float[] orig_im_size_values = { (float)orgHei, (float)orgWid };
165 var orig_im_size_values_tensor = new DenseTensor<float>(orig_im_size_values, new[] { 2 });
166
167 var decode_inputs = new List<NamedOnnxValue>
168 {
169 NamedOnnxValue.CreateFromTensor("image_embeddings", embedding_tensor),
170 NamedOnnxValue.CreateFromTensor("point_coords", point_coords_tensor),
171 NamedOnnxValue.CreateFromTensor("point_labels", point_label_tensor),
172 NamedOnnxValue.CreateFromTensor("mask_input", mask_tensor),
173 NamedOnnxValue.CreateFromTensor("has_mask_input", hasMaskValues_tensor),
174 NamedOnnxValue.CreateFromTensor("orig_im_size", orig_im_size_values_tensor)
175 };
176
177 var segmask = this.mDecoder.Run(decode_inputs);
178 var outputmask = segmask.First().AsTensor<float>().ToArray();
179 return outputmask;
