84 public override byte[] GetTile(TileInfo tileInfo)

97 var bgraData = SlideImage.ReadRegion(tileInfo.Index.Level, (long)curLevelOffsetXPixel, (long)curLevelOffsetYPixel, curTileWidth, curTileHeight);
98 //We check to see if the data is valid.
99 if (bgraData.Length != curTileWidth * curTileHeight * 4)
100 return null;
101 byte[] bm = ConvertRgbaToRgb(bgraData);
102 if (_enableCache && bgraData != null)

106 public static byte[] ConvertRgbaToRgb(byte[] rgbaArray)
107 {
108 // Initialize a new byte array for RGB24 format
109 byte[] rgbArray = new byte[(rgbaArray.Length / 4) * 3];
110
111 for (int i = 0, j = 0; i < rgbaArray.Length; i += 4, j += 3)
112 {
113 // Copy the R, G, B values, skip the A value
114 rgbArray[j] = rgbaArray[i + 2]; // B
115 rgbArray[j + 1] = rgbaArray[i + 1]; // G
116 rgbArray[j + 2] = rgbaArray[i]; // R
117 }
118
119 return rgbArray;
120 }
