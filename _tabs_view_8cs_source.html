148 protected TabsView(Builder builder, IntPtr handle) : base(handle)

153 filteredMenu.Active = true;

157 new Gtk.FileChooserDialog("Choose file to open",
158 this,
159 FileChooserAction.Open,
160 "Cancel", ResponseType.Cancel,
161 "OK", ResponseType.Accept);

164 foreach (RotateFlipType flip in (RotateFlipType[])Enum.GetValues(typeof(RotateFlipType)))

166 MenuItem mi = new MenuItem(flip.ToString());

168 m.Append(mi);

171 rotateFlipMenu.ShowAll();

304 private void ExportROIsFromImageJMenu_ButtonPressEvent(object o, ButtonPressEventArgs args)
305 {
306 Gtk.FileChooserDialog filechooser =
307 new Gtk.FileChooserDialog("Set ROI filename to save",
308 this,
309 FileChooserAction.Save,
310 "Cancel", ResponseType.Cancel,
311 "Save", ResponseType.Accept);
312 if (filechooser.Run() != (int)ResponseType.Accept)
313 return;
314 int i = 1;
315 foreach (ROI item in ImageView.SelectedImage.Annotations)
316 {
317 string f = filechooser.Filename;
318 if (!f.EndsWith(".roi"))
319 f += i + ".roi";
320 else
321 {
322 f = f.Replace(".roi","") + i + ".roi";
323 }
324 ImageJ.RoiEncoder.save(item, f);
325 i++;
326 }
327 filechooser.Hide();
328 }

362 private void TabsView_WindowStateEvent(object o, WindowStateEventArgs args)
363 {
364 if ((args.Event.ChangedMask & Gdk.WindowState.Iconified) != 0)
365 {
366 foreach (ImageView v in viewers)
367 {
368 v.Hide();
369 }
370 }
371
372 else if ((args.Event.ChangedMask & Gdk.WindowState.Maximized) == 0 && (args.Event.ChangedMask & Gdk.WindowState.Iconified) == 0)
373 {
374 foreach (ImageView v in viewers)
375 {
376 if(!v.Visible)
377 v.Present();
378 }
379 }
