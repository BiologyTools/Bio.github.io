63 public virtual byte[] GetSlice(SliceInfo sliceInfo)
64 {
65 var curLevel = ImageView.SelectedImage.Level;
66 var curUnitsPerPixel = Schema.Resolutions[curLevel].UnitsPerPixel;
67 var tileInfos = Schema.GetTileInfos(sliceInfo.Extent, curLevel);
68 List<Tuple<Extent, byte[]>> tiles = new List<Tuple<Extent, byte[]>>();
69 foreach (TileInfo t in tileInfos)
70 {
71 byte[] cache = null;
72 if (_bgraCache.ContainsKey(t.Index))
73 {
74 cache = _bgraCache[t.Index];
75 }
76 if (cache == null)
77 {
78 cache = GetTile(t);
79 _bgraCache.Add(t.Index, cache);
80 }
81 tiles.Add(Tuple.Create(t.Extent.WorldToPixelInvertedY(curUnitsPerPixel), cache));
82 }
83 var srcPixelExtent = sliceInfo.Extent.WorldToPixelInvertedY(curUnitsPerPixel);
84 var dstPixelExtent = sliceInfo.Extent.WorldToPixelInvertedY(sliceInfo.Resolution);
85 var dstPixelHeight = sliceInfo.Parame.DstPixelHeight > 0 ? sliceInfo.Parame.DstPixelHeight : dstPixelExtent.Height;
86 var dstPixelWidth = sliceInfo.Parame.DstPixelWidth > 0 ? sliceInfo.Parame.DstPixelWidth : dstPixelExtent.Width;
87 destExtent = new Extent(0, 0, dstPixelWidth, dstPixelHeight);
88 sourceExtent = srcPixelExtent;
89 if (UseVips)
90 {
91 try
92 {
93 NetVips.Image im = OpenSlideGTK.ImageUtil.JoinVips(tiles, srcPixelExtent, new Extent(0, 0, dstPixelWidth, dstPixelHeight));
94 LastSlice = im.WriteToMemory();
95 return LastSlice;
96 }
97 catch (Exception e)
98 {
99 UseVips = false;
100 Console.WriteLine("Failed to use LibVips please install Libvips for your platform.");
101 Console.WriteLine(e.Message);
102 }
103 }
104 try
105 {
106 Image<Rgb24> im = OpenSlideGTK.ImageUtil.Join(tiles, srcPixelExtent, new Extent(0, 0, dstPixelWidth, dstPixelHeight));
107 LastSlice = GetRgb24Bytes(im);
108 im.Dispose();
109 }
110 catch (Exception er)
111 {
112 Console.WriteLine(er.Message);
113 return null;
114 }
115 return LastSlice;
116 }
117 public byte[] GetRgb24Bytes(Image<Rgb24> image)
118 {
119 int width = image.Width;
120 int height = image.Height;
121 byte[] rgbBytes = new byte[width * height * 3]; // 3 bytes per pixel (RGB)
122
123 int byteIndex = 0;
124 for (int y = 0; y < height; y++)
125 {
126 for (int x = 0; x < width; x++)
127 {
128 Rgb24 pixel = image[x, y];
129 rgbBytes[byteIndex++] = pixel.B;
130 rgbBytes[byteIndex++] = pixel.G;
131 rgbBytes[byteIndex++] = pixel.R;
132 }
133 }
134
135 return rgbBytes;
136 }
