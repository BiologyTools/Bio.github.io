66 public void Encode(BioImage b)
67 {
68 if (b.Buffers.Count * 3 * 1024 * 1024 > 4e8)
69 {
70 // Create the message dialog
71 MessageDialog msgBox = new MessageDialog(
72 null,
73 DialogFlags.Modal,
74 MessageType.Info,
75 ButtonsType.Ok | ButtonsType.Cancel,
76 "Memory required is more than 4GB are you sure you want to continue?"
77 );
78
79 // Show the message dialog
80 if (msgBox.Run() != (int)ResponseType.Ok)
81 return;
82 }
83 Progress pr = Progress.Create("Encoding Image", "Transforming", "");
84 // update ui on main UI thread
85 Application.Invoke(delegate
86 {
87 pr.Show();
88 pr.Present();
89 });
90 int i = 0;
91 foreach (Bitmap bu in b.Buffers)
92 {
93 Transforms tranform = new Transforms(1024);
94 float[] img = tranform.ApplyImage(bu);
95 var tensor = new DenseTensor<float>(img, new[] { 1, 3, 1024, 1024 });
96 IDisposableReadOnlyCollection<DisposableNamedOnnxValue> results = null;
97 List<NamedOnnxValue> inputs = new List<NamedOnnxValue>
98 {
99 NamedOnnxValue.CreateFromTensor("image", tensor)
100 };
101 try
102 {
103 results = this.mEncoder.Run(inputs);
104 }
105 catch (Exception e)
106 {
107 Console.WriteLine("Tried running SAM-2 next trying SAM-1");
108 SAM2 = false;
109 }
110 if(!SAM2)
111 {
112 inputs = new List<NamedOnnxValue>
113 {
114 NamedOnnxValue.CreateFromTensor("x", tensor)
115 };
116 results = this.mEncoder.Run(inputs);
117 }
118 if (b.Tag == null)
119 {
120 b.Tag = new List<float[]>();
121 }
122 List<float[]> l = (List<float[]>)b.Tag;
123 if(!SAM2)
124 l.Add(results.First().AsTensor<float>().ToArray());
125 else
126 l.Add(results.Last().AsTensor<float>().ToArray());
127 b.Tag = l;
128 results.Dispose();
129 pr.ProgressValue = (double)i/b.Buffers.Count;
130 i++;
131 }
132 // update ui on main UI thread
133 Application.Invoke(delegate
134 {
135 pr.Hide();
136 });
137 this.mReady = true;
138 }

227 decode_inputs = new List<NamedOnnxValue>
228 {
229 NamedOnnxValue.CreateFromTensor("image_embeddings", embedding_tensor),
230 NamedOnnxValue.CreateFromTensor("point_coords", point_coords_tensor),
231 NamedOnnxValue.CreateFromTensor("point_labels", point_label_tensor),
