163 protected TabsView(Builder builder, IntPtr handle) : base(handle)
164 {
165 _builder = builder;
166 App.tabsView = this;
167 builder.Autoconnect(this);
168 filteredMenu.Active = true;
169 SetupHandlers();
170 Function.InitializeMainMenu();
171 filechooser =
172 new Gtk.FileChooserDialog("Choose file to open",
173 this,
174 FileChooserAction.Open,
175 "Cancel", ResponseType.Cancel,
176 "OK", ResponseType.Accept);
177 filechooser.SelectMultiple = true;
178 Menu m = new Menu();
179 foreach (RotateFlipType flip in (RotateFlipType[])Enum.GetValues(typeof(RotateFlipType)))
180 {
181 MenuItem mi = new MenuItem(flip.ToString());
182 mi.ButtonPressEvent += RotateFlip_ButtonPressEvent;
183 m.Append(mi);
184 }
185 rotateFlipMenu.Submenu = m;
186 rotateFlipMenu.ShowAll();
187 string a = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
188 Menu me = new Menu();
189 foreach (char c in a)
190 {
191 MenuItem mi = new MenuItem(c.ToString());
192 Menu men = new Menu();
193 foreach (ImageJ.Macro.Command command in ImageJ.Macro.Commands.Values)
194 {
195 if (command.Name.StartsWith(c))
196 {
197 MenuItem menuItem = new MenuItem(command.Name.ToString());
198 menuItem.ButtonPressEvent += CommandMenuItem_ButtonPressEvent;
199 men.Append(menuItem);
200 }
201 }
202 mi.Submenu = men;
203 me.Append(mi);
204 }
205 commandsMenu.Submenu = me;
206 commandsMenu.ShowAll();
207 recentMenu.Submenu = recent;
208 App.ApplyStyles(this);
209 Menu rm = new Menu();
210 foreach(Scripting.Script s in Scripting.scripts.Values)
211 {
212 MenuItem mi = new MenuItem(s.name);
213 mi.ButtonPressEvent += Mi_ButtonPressEvent;
214 rm.Append(mi);
215 }
216 foreach (ImageJ.Macro.Command c in ImageJ.Macros)
217 {
218 MenuItem mi = new MenuItem(c.Name);
219 mi.ButtonPressEvent += Mi_ButtonPressEvent;
220 rm.Append(mi);
221 }
222 runMenu.Submenu = rm;
223 runMenu.ShowAll();
224 Plugins.Initialize();
225 }

227 private void Mi_ButtonPressEvent(object o, ButtonPressEventArgs args)
228 {
229 MenuItem m = (MenuItem)o;
230 if (m.Label.EndsWith(".ijm") || m.Label.EndsWith(".txt") && !m.Label.EndsWith(".cs"))
231 {
232 string ma = File.ReadAllText(m.Label);
233 ImageJ.RunOnImage(ma, BioConsole.headless, BioConsole.onTab, BioConsole.useBioformats, BioConsole.resultInNewTab);
234 }
235 else
236 Scripting.RunByName(m.Label);
237 }

239 private void CommandMenuItem_ButtonPressEvent(object o, ButtonPressEventArgs args)
240 {
241 if (ImageView.SelectedImage == null) return;
242 MenuItem m = (MenuItem)o;
243 ImageJ.RunOnImage("run(\"" + m.Label + "\");", BioConsole.headless, BioConsole.onTab, BioConsole.useBioformats, BioConsole.resultInNewTab);
244 MenuItem mi = new MenuItem(m.Label);
245 mi.ButtonPressEvent += CommandMenuItem_ButtonPressEvent;
246 bool con = false;
247 foreach (MenuItem item in recent.Children)
248 {
249 if (item.Label == m.Label)
250 con = true;
251 }
252 if(!con)
253 recent.Append(mi);
254 recentMenu.ShowAll();
255 }

435 private void ExportROIsFromImageJMenu_ButtonPressEvent(object o, ButtonPressEventArgs args)
436 {
437 Gtk.FileChooserDialog filechooser =
438 new Gtk.FileChooserDialog("Set ROI filename to save",
439 this,
440 FileChooserAction.Save,
441 "Cancel", ResponseType.Cancel,
442 "Save", ResponseType.Accept);
443 if (filechooser.Run() != (int)ResponseType.Accept)
444 return;
445 int i = 1;
446 foreach (ROI item in ImageView.SelectedImage.Annotations)
447 {
448 string f = filechooser.Filename;
449 if (!f.EndsWith(".roi"))
450 f += i + ".roi";
451 else
452 {
453 f = f.Replace(".roi","") + i + ".roi";
454 }
455 ImageJ.RoiEncoder.save(item, f);
456 i++;
457 }
458 filechooser.Hide();
459 }

493 private void TabsView_WindowStateEvent(object o, WindowStateEventArgs args)
494 {
495 if ((args.Event.ChangedMask & Gdk.WindowState.Iconified) != 0)
496 {
497 foreach (ImageView v in viewers)
498 {
499 v.Hide();
500 }
501 }
502
503 else if ((args.Event.ChangedMask & Gdk.WindowState.Maximized) == 0 && (args.Event.ChangedMask & Gdk.WindowState.Iconified) == 0)
504 {
505 foreach (ImageView v in viewers)
506 {
507 if(!v.Visible)
508 v.Present();
509 }
510 }
