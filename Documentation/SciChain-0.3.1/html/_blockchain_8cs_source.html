663 public static bool ProcessTransaction(Transaction transaction)
664 {
665 //If this transaction has already been processed we skip and return false.)
666 if(transaction.Signature != null)
667 if (GetTransaction(transaction) != null)
668 {
669 Console.WriteLine("Processing Transaction: Skipping already processed transaction.");
670 return false;
671 }
672 Console.WriteLine("Processing Transaction:" + transaction.TransactionType);
673 if (!VerifyTransaction(transaction))
674 {
675 Console.WriteLine("Transaction failed: Not a valid transaction.");
676 return false;
677 }
678 Directory.CreateDirectory(dir + "/PendingTransactions");
679
680 if (transaction.TransactionType == Transaction.Type.transaction)
681 {
682 // Proceed with adding the transaction
683 if (transaction.FromAddress != null)
684 {
685 var senderBalance = GetBalance(transaction.FromAddress);
686
687 if (senderBalance < transaction.Amount)
688 {
689 Console.WriteLine("Transaction failed: Not enough balance.");
690 return false;
691 }
692 }
693 SavePendingTransaction(transaction);
694 }
695 else if (transaction.TransactionType == Transaction.Type.registration)
696 {
697 transaction.Amount = gift;
698 //We will register this user and associate their ORCID ID with their public key (RSAParameters)
699 //If this is a new user we will send them the gift transaction from the treasury.
700 bool found = false;
701 foreach (Block item in Chain)
702 {
703 if(item.Transactions!=null && item.Transactions.Count > 0)
704 foreach (Transaction t in item.Transactions)
705 {
706 if(t.TransactionType == Transaction.Type.registration && t.ToAddress == transaction.ToAddress)
707 {
708 found = true;
709 break;
710 }
711 }
712 if (found)
713 break;
714 }
715 if(!found)
716 {
717 treasuryBalance -= gift;
718 SavePendingTransaction(transaction);
719 }
720 }
721 else if (transaction.TransactionType == Transaction.Type.review)
722 {
723 transaction.Amount = gift;
724 int revs = GetReviews(transaction.Data);
725 int fls = GetFlags(transaction.Data);
726 Console.WriteLine("Block: " + transaction.Data + " Reviews: " + revs + " Flags: " + fls);
727 SavePendingTransaction(transaction);
728 if(revs >= reviewers && fls <= flags)
729 {
730 MineBlock(transaction.Data);
731 }
732 }
733 else if (transaction.TransactionType == Transaction.Type.flag)
734 {
735 transaction.Amount = gift;
736 SavePendingTransaction(transaction);
737 }
738 else if (transaction.TransactionType == Transaction.Type.blockreward)
739 {
740 transaction.Amount = miningReward;
741 SavePendingTransaction(transaction);
742 }
743 else if (transaction.TransactionType == Transaction.Type.addreputation)
744 {
745 transaction.Amount = 1;
746 SavePendingTransaction(transaction);
747 }
748 else if (transaction.TransactionType == Transaction.Type.removereputation)
749 {
750 transaction.Amount = 10;
751 SavePendingTransaction(transaction);
752 }
753 return true;
754 }
766 public static bool VerifyTransaction(Transaction transaction)
767 {
768 using (var rsa = new RSACryptoServiceProvider())
769 {
770 try
771 {
772 rsa.ImportParameters(RSA.StringToRSAParameters(transaction.PublicKey));
773 string dataToVerify;
774 if(transaction.FromAddress == null || transaction.FromAddress == "")
775 dataToVerify = transaction.ToAddress + Transaction.DecimalToStringWithMaxDecimals(transaction.Amount);
776 else
777 dataToVerify = transaction.FromAddress + transaction.ToAddress + Transaction.DecimalToStringWithMaxDecimals(transaction.Amount);
778 var dataToVerifyBytes = Encoding.UTF8.GetBytes(dataToVerify);
779 var hasher = new SHA256Managed();
780 var hashedData = hasher.ComputeHash(dataToVerifyBytes);
781 return rsa.VerifyData(hashedData, CryptoConfig.MapNameToOID("SHA256"), Convert.FromBase64String(transaction.Signature));
782 }
783 catch
784 {
785 return false;
786 }
787 }

880 public static void ProcessMessage(Message message,Peer peer)
881 {
882 Console.WriteLine("Processsing message. " + message.Type);
883 if(peer == null)
884 {
885 Console.WriteLine("ProcessMessage: Peer is null");
886 return;
887 }
888 string con = message.Content;
889 if (message.Type == "NewBlock")
890 {
891 var block = JsonConvert.DeserializeObject<Block>(con);
892 if (block == null)
893 loaded = true;

895 {
896 AddBlock(block);

898 }
899 else
900 loaded = true;
901 }
902 else
903 if (message.Type == "NewTransaction")
904 {

910 {

912 if (Peers.Count < maxPeers)

921 }

924 ConnectToPeer(pr.Address, new ChatClient(pr.Address, port), port);
925 }
926 BroadcastNewPeer(pr);
927 Console.WriteLine(message.Type + " " + pr.Address);
928 }
929 else
930 if(message.Type == "Peers")
931 {
932 var prs = JsonConvert.DeserializeObject<Peer[]>(con);
933 foreach (var pr in prs)
934 {
935 if (Peers.Count < maxPeers)
936 {
937 bool ex = false;
938 foreach (var p in Peers)
939 {
940 if (pr.Address == p.Value.Address)
941 {
942 ex = true;
943 break;
944 }
945 }

947 ConnectToPeer(pr.Address, new ChatClient(pr.Address, port), port);
948 }
949 }
950
951 Console.WriteLine(message.Type);
952 }
953 else
954 if(message.Type == "GetBlock")
