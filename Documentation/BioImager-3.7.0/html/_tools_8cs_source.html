448 ImageView.selectedAnnotations.Clear();
449 RectangleD r = GetTool(Tool.Type.rectSel).Rectangle;
450 foreach (ROI an in App.viewer.AnnotationsRGB)
451 {
452 if (an.GetSelectBound(App.viewer.GetScale(), App.viewer.GetScale()).ToRectangleF().IntersectsWith(r.ToRectangleF()))
453 {

457 RectangleD[] sels = an.GetSelectBoxes(selectBoxSize);
458 for (int i = 0; i < sels.Length; i++)

471 else
472 if (Tools.currentTool.type == Tools.Tool.Type.magic && buts == MouseButtons.Left)
473 {
474 PointD pf = new PointD(ImageView.mouseUp.X - ImageView.mouseDown.X, ImageView.mouseUp.Y - ImageView.mouseDown.Y);
475 ZCT coord = App.viewer.GetCoordinate();
476
477 Rectangle r = new Rectangle((int)ImageView.mouseDown.X, (int)ImageView.mouseDown.Y, (int)(ImageView.mouseUp.X - ImageView.mouseDown.X), (int)(ImageView.mouseUp.Y - ImageView.mouseDown.Y));
478 if (r.Width <= 2 || r.Height <= 2)
479 return;
480 AForge.Bitmap bf = ImageView.SelectedImage.Buffers[ImageView.SelectedImage.Coords[coord.Z, coord.C, coord.T]].GetCropBuffer(r);
481 Statistics[] sts = Statistics.FromBytes(bf);
482 Statistics st = sts[0];
483 Threshold th;
484 if (magicSel.Numeric)
485 {
486 th = new Threshold((int)magicSel.Threshold);
487 }
488 else
489 if (magicSel.Index == 2)
490 th = new Threshold((int)(st.Min + st.Mean));
491 else
492 if (magicSel.Index == 1)
493 th = new Threshold((int)st.Median);
494 else
495 th = new Threshold(st.Min);
496 th.ApplyInPlace(bf);
497 Invert inv = new Invert();
498 AForge.Bitmap det;
499 if (bf.BitsPerPixel > 8)
500 det = AForge.Imaging.Image.Convert16bppTo8bpp(bf);
501 else
502 det = bf;
503 BlobCounter blobCounter = new BlobCounter();
504 blobCounter.ProcessImage(det);
505 Blob[] blobs = blobCounter.GetObjectsInformation();
506 // create convex hull searching algorithm
507 GrahamConvexHull hullFinder = new GrahamConvexHull();
508 // lock image to draw on it
509 // process each blob
510 foreach (Blob blob in blobs)
511 {
512 if (blob.Rectangle.Width < magicSel.Max && blob.Rectangle.Height < magicSel.Max)
513 continue;
514 List<IntPoint> leftPoints = new List<IntPoint>();
515 List<IntPoint> rightPoints = new List<IntPoint>();
516 List<IntPoint> edgePoints = new List<IntPoint>();
517 List<IntPoint> hull = new List<IntPoint>();
518 // get blob's edge points
519 blobCounter.GetBlobsLeftAndRightEdges(blob,
520 out leftPoints, out rightPoints);
521 edgePoints.AddRange(leftPoints);
522 edgePoints.AddRange(rightPoints);
523 // blob's convex hull
524 hull = hullFinder.FindHull(edgePoints);
525 PointD[] pfs = new PointD[hull.Count];
526 for (int i = 0; i < hull.Count; i++)
527 {
528 pfs[i] = new PointD(r.X + hull[i].X, r.Y + hull[i].Y);
529 }
530 ROI an = ROI.CreateFreeform(coord, pfs);
531 ImageView.SelectedImage.Annotations.Add(an);
532 }
533 }
534 else
535 if (Tools.currentTool.type == Tools.Tool.Type.bucket && buts == MouseButtons.Left)
536 {
537 ZCT coord = App.viewer.GetCoordinate();
538 floodFiller.FillColor = DrawColor;
539 floodFiller.Tolerance = currentTool.tolerance;
540 floodFiller.Bitmap = ImageView.SelectedImage.Buffers[ImageView.SelectedImage.Coords[coord.C, coord.Z, coord.T]];
541 floodFiller.FloodFill(new AForge.Point((int)p.X, (int)p.Y));
542 App.viewer.UpdateImages();
543 }
544 else
545 if (Tools.currentTool.type == Tools.Tool.Type.dropper && buts == MouseButtons.Left)
546 {
547 DrawColor = ImageView.SelectedBuffer.GetPixel((int)p.X,(int)p.Y);
548 UpdateGUI();
549 }
