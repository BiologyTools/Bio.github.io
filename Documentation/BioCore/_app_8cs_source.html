204 private static IEnumerable<ToolStripMenuItem> GetItems(ToolStripMenuItem item)
205 {
206 foreach (var dropDownItem in item.DropDownItems)
207 {
208 if (dropDownItem.GetType() == typeof(ToolStripMenuItem))
209 {
210 if (((ToolStripMenuItem)dropDownItem).HasDropDownItems)
211 {
212 foreach (ToolStripMenuItem subItem in GetItems((ToolStripMenuItem)dropDownItem))
213 yield return subItem;
214 }
215 yield return (ToolStripMenuItem)dropDownItem;
216 }
217 }

377 private static void ItemClicked(object sender, EventArgs e)
378 {
379 ToolStripMenuItem ts = (ToolStripMenuItem)sender;
380 if (ts.Text.EndsWith(".dll"))
381 {
382 Plugin.Plugins[ts.Text].Execute(new string[] { });
383 }
384 else if (ts.Text.EndsWith(".ijm"))
385 {
386 string s = File.ReadAllText(Path.GetDirectoryName(ImageJ.ImageJPath) + "/macros/" + ts.Text);
387 ImageJ.RunOnImage(s, BioConsole.headless, BioConsole.onTab, BioConsole.useBioformats, BioConsole.newTab);
388 }
389 else
390 {
391 Function f = (Function)ts.Tag;
392 f.PerformFunction(true);
