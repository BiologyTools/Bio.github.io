201 private void RoiView_RowActivated(object o, RowActivatedArgs args)
202 {
203 TreeIter iter;
204 if (roiView.Model.GetIter(out iter, args.Path))
205 {
206 string type = (string)roiView.Model.GetValue(iter, 0);
207 ROI.Type t;
208 if (Enum.TryParse<ROI.Type>(type,out t))
209 {
210 string id = (string)roiView.Model.GetValue(iter, 1);
211 string text = (string)roiView.Model.GetValue(iter, 2);
212 string bounds = (string)roiView.Model.GetValue(iter, 3);
213 if (rois.ContainsKey(type.ToString() + "," + id + "," + text + "," + bounds))
214 {
215 anno = rois[type.ToString() + "," + id + "," + text + "," + bounds];
216 if (App.viewer != null)
217 App.viewer.SetCoordinate(anno.coord.Z, anno.coord.C, anno.coord.T);
218
219 if (anno.type == ROI.Type.Line || anno.type == ROI.Type.Polygon ||
220 anno.type == ROI.Type.Polyline)
221 {
222 xBox.Sensitive = false;
223 yBox.Sensitive = false;
224 wBox.Sensitive = false;
225 hBox.Sensitive = false;
226 }
227 else
228 {
229 xBox.Sensitive = true;
230 yBox.Sensitive = true;
231 wBox.Sensitive = true;
232 hBox.Sensitive = true;
233 }
234 if (anno.type == ROI.Type.Rectangle || anno.type == ROI.Type.Ellipse)
235 {
236 pointBox.Sensitive = false;
237 pointXBox.Sensitive = false;
238 pointYBox.Sensitive = false;
239 }
240 else
241 {
242 pointBox.Sensitive = true;
243 pointXBox.Sensitive = true;
244 pointYBox.Sensitive = true;
245 }
246 xBox.Value = anno.X;
247 yBox.Value = anno.Y;
248 wBox.Value = anno.W;
249 hBox.Value = anno.H;
250 zBox.Value = anno.coord.Z;
251 cBox.Value = anno.coord.C;
252 tBox.Value = anno.coord.T;
253 rBox.Value = anno.strokeColor.R;
254 gBox.Value = anno.strokeColor.G;
255 bBox.Value = anno.strokeColor.B;
256 widthBox.Value = anno.strokeWidth;
257 idBox.Text = anno.id;
258 textBox.Text = anno.Text;
259 typeLabel.Text = anno.type.ToString();
260 UpdatePointBox();
261 if(ImageView.SelectedImage.isPyramidal)
262 App.viewer.PyramidalOrigin = new PointD((int)(anno.X), (int)(anno.Y));
263 }
264 else return;
265 }
266 else return;
267 }
268 else

322 public void UpdateAnnotationList()
323 {
324 rois.Clear();

326 foreach (BioImage b in Images.images)
