161 public static ToolStripItem GetMenuItemFromPath(string s, Function f)
162 {
163 if (s == "" || s == null)
164 return null;
165 string[] sts = s.Split('/');
166 start:
167 List<ToolStripMenuItem> allItems = GetMenuItems();
168 //Find path or create it.
169 bool found = false;
170 ToolStripMenuItem item = null;
171
172 for (int t = 0; t < sts.Length; t++)
173 {
174 found = false;
175 for (int i = 0; i < allItems.Count; i++)
176 {
177 if (allItems[i].Text == sts[t])
178 {
179 item = allItems[i];
180 found = true;
181 if (t == sts.Length - 1)
182 return allItems[i];
183 }
184 }
185 if (!found)
186 {
187 if(t == 0)
188 {
189 tabsView.MainMenuStrip.Items.Add(sts[t]);
190 goto start;
191 }
192 else if(t > 0 && t < sts.Length)
193 {
194 ToolStripMenuItem itm = new ToolStripMenuItem();
195 itm.Tag = f;
196 itm.Name = f.Name;
197 item.DropDownItems.Add(f.Name, null, ItemClicked);
198 return item;
199 }
200 else
201 {
202 item.DropDownItems.Add(sts[t]);
203 }
204 }
205 }
206 return item;
207 }
