60 /* A constructor for a class called TabsView. It is initializing the component and loading
61 properties. It is also initializing the class. */
62 public TabsView(string[] arg)
63 {
64 InitializeComponent();
65 LoadProperties();
66 Init();
67 if (arg.Length == 0)
68 return;
69 else
70 {
71 for (int i = 0; i < arg.Length; i++)
72 {
73 if (arg[i].EndsWith(".cs"))
74 {
75 App.runner.RunScriptFile(arg[0]);
76 return;
77 }
78 if (arg[i].EndsWith(".ijm"))
79 {
80 ImageJ.RunMacro(arg[i], "");
81 return;
82 }
83 else
84 {
85 BioImage.OpenFile(arg[i],true);
86 }
87 }
88 }

96 public void AddTab(BioImage b)
97 {
98 if (b == null)
99 return;
100 if (b.filename.Contains("/") || b.filename.Contains("\\"))
101 b.filename = Path.GetFileName(b.filename);
102 TabPage t = new TabPage(b.filename);
103 ImageView v = new ImageView(b);
104 v.Dock = DockStyle.Fill;
105 t.Controls.Add(v);
106 if (Width < b.SizeX || Height < b.SizeY)
107 {
108 Width = b.SizeX;
109 Height = b.SizeY + 190;
110 }
111 tabControl.TabPages.Add(t);
112 tabControl.Dock = DockStyle.Fill;
113 tabControl.SelectedIndex = tabControl.TabCount - 1;
114 ResizeView();
115 }

850 private void TabsView_KeyDown(object sender, KeyEventArgs e)
851 {
852 if (Viewer == null)
853 return;
854 double moveAmount = 1500 * (1 / App.viewer.Scale.Width);
855 if (e.KeyCode == Keys.C && e.Control)
856 {
857 Viewer.CopySelection();
858 return;
859 }
860 if (e.KeyCode == Keys.V && e.Control)
861 {
862 Viewer.PasteSelection();
863 return;
864 }
865 if (e.KeyCode == Keys.Subtract || e.KeyCode == Keys.NumPad7)
866 {
867 App.viewer.Scale = new AForge.SizeF(App.viewer.Scale.Width - 0.1f, App.viewer.Scale.Height - 0.1f);
868 }
869 if (e.KeyCode == Keys.Add || e.KeyCode == Keys.NumPad9)
870 {
871 App.viewer.Scale = new AForge.SizeF(App.viewer.Scale.Width - 0.1f, App.viewer.Scale.Height - 0.1f);
872 }
873 if (e.KeyCode == Keys.W || e.KeyCode == Keys.NumPad8)
874 {
875 Viewer.Origin = new PointD(Viewer.Origin.X, Viewer.Origin.Y - moveAmount);
876 }
877 if (e.KeyCode == Keys.S || e.KeyCode == Keys.NumPad2)
878 {
879 Viewer.Origin = new PointD(Viewer.Origin.X, Viewer.Origin.Y + moveAmount);
880 }
881 if (e.KeyCode == Keys.A || e.KeyCode == Keys.NumPad4)
882 {
883 Viewer.Origin = new PointD(Viewer.Origin.X - moveAmount, Viewer.Origin.Y);
884 }
885 if (e.KeyCode == Keys.D || e.KeyCode == Keys.NumPad6)
886 {
887 Viewer.Origin = new PointD(Viewer.Origin.X + moveAmount, Viewer.Origin.Y);
888 }
889 if (e.KeyCode == Keys.Up)
890 Microscope.MoveUp(Microscope.viewSize.Y / 2);
891 if (e.KeyCode == Keys.Down)
892 Microscope.MoveDown(Microscope.viewSize.Y / 2);
893 if (e.KeyCode == Keys.Left)
894 Microscope.MoveRight(-(Microscope.viewSize.X / 2));
895 if (e.KeyCode == Keys.Right)
896 Microscope.MoveRight((Microscope.viewSize.X / 2) / 2);
897 Viewer.UpdateStatus();
898 Viewer.UpdateView();
899 }

1067 return;
1068 }
