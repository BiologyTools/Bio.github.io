531 if (Tools.currentTool.type == Tools.Tool.Type.magic && buts.Event.Button == 1)
532 {
533 PointD pf = new PointD(ImageView.mouseUp.X - ImageView.mouseDown.X, ImageView.mouseUp.Y - ImageView.mouseDown.Y);
534 ZCT coord = App.viewer.GetCoordinate();
535
536 Rectangle r = new Rectangle((int)ImageView.mouseDown.X, (int)ImageView.mouseDown.Y, (int)(ImageView.mouseUp.X - ImageView.mouseDown.X), (int)(ImageView.mouseUp.Y - ImageView.mouseDown.Y));
537 if (r.Width <= 2 || r.Height <= 2)
538 return;
539 AForge.Bitmap bf = ImageView.SelectedImage.Buffers[ImageView.SelectedImage.Coords[coord.Z, coord.C, coord.T]].GetCropBuffer(r);
540 Statistics[] sts = Statistics.FromBytes(bf);
541 Statistics st = sts[0];
542 AForge.Bitmap crop = bf;
543 Threshold th;
544 if (magicSel.Numeric)
545 {
546 th = new Threshold((int)magicSel.Threshold);
547 }
548 else
549 if (magicSel.Index == 2)
550 th = new Threshold((int)(st.Min + st.Mean));
551 else
552 if (magicSel.Index == 1)
553 th = new Threshold((int)st.Median);
554 else
555 th = new Threshold(st.Min);
556 th.ApplyInPlace(crop);
557 Invert inv = new Invert();
558 AForge.Bitmap det;
559 if (bf.BitsPerPixel > 8)
560 det = AForge.Imaging.Image.Convert16bppTo8bpp((crop));
561 else
562 det = crop;
563 BlobCounter blobCounter = new BlobCounter();
564 blobCounter.ProcessImage(det);
565 Blob[] blobs = blobCounter.GetObjectsInformation();
566 // create convex hull searching algorithm
567 GrahamConvexHull hullFinder = new GrahamConvexHull();
568 // lock image to draw on it
569 // process each blob
570 foreach (Blob blob in blobs)
571 {
572 if (blob.Rectangle.Width < magicSel.Max && blob.Rectangle.Height < magicSel.Max)
573 continue;
574 List<IntPoint> leftPoints = new List<IntPoint>();
575 List<IntPoint> rightPoints = new List<IntPoint>();
576 List<IntPoint> edgePoints = new List<IntPoint>();
577 List<IntPoint> hull = new List<IntPoint>();
578 // get blob's edge points
579 blobCounter.GetBlobsLeftAndRightEdges(blob,
580 out leftPoints, out rightPoints);
581 edgePoints.AddRange(leftPoints);
582 edgePoints.AddRange(rightPoints);
583 // blob's convex hull
584 hull = hullFinder.FindHull(edgePoints);
585 PointD[] pfs = new PointD[hull.Count];
586 for (int i = 0; i < hull.Count; i++)
587 {
588 pfs[i] = new PointD(r.X + hull[i].X, r.Y + hull[i].Y);
589 }
590 ROI an = ROI.CreateFreeform(coord, pfs);
591 ImageView.SelectedImage.Annotations.Add(an);
592 }
593 }
594 else
595 if (Tools.currentTool.type == Tools.Tool.Type.bucket && buts.Event.Button == 1)
596 {
597 if (mouseU.X >= ImageView.SelectedImage.SizeX && mouseU.Y >= ImageView.SelectedImage.SizeY)
598 return;
599 floodFiller.FillColor = DrawColor;
600 floodFiller.Tolerance = new ColorS(0, 0, 0);
601 floodFiller.Bitmap = ImageView.SelectedBuffer;
602 floodFiller.FloodFill(new AForge.Point((int)mouseU.X, (int)mouseU.Y));

604 }
