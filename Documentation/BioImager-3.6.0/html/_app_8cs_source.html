166 public static ToolStripItem GetMenuItemFromPath(string s, Function f)
167 {
168 if (s == "" || s == null)
169 return null;
170 string[] sts = s.Split('/');
171 start:
172 List<ToolStripMenuItem> allItems = GetMenuItems();
173 //Find path or create it.
174 bool found = false;
175 ToolStripMenuItem item = null;
176
177 for (int t = 0; t < sts.Length; t++)
178 {
179 found = false;
180 for (int i = 0; i < allItems.Count; i++)
181 {
182 if (allItems[i].Text == sts[t])
183 {
184 item = allItems[i];
185 found = true;
186 if (t == sts.Length - 1)
187 return allItems[i];
188 }
189 }
190 if (!found)
191 {
192 if(t == 0)
193 {
194 tabsView.MainMenuStrip.Items.Add(sts[t]);
195 goto start;
196 }
197 else if(t > 0 && t < sts.Length)
198 {
199 ToolStripMenuItem itm = new ToolStripMenuItem();
200 itm.Tag = f;
201 itm.Name = f.Name;
202 item.DropDownItems.Add(f.Name, null, ItemClicked);
203 return item;
204 }
205 else
206 {
207 item.DropDownItems.Add(sts[t]);
208 }
209 }
210 }
211 return item;
212 }
