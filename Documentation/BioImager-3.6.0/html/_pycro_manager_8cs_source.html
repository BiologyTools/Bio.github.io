151 private static bool TakeImage(string file, out int width, out int height, out int depth, out int bits, out int channels, out AForge.Bitmap[] bm)
152 {
153 string fil = Path.GetFileNameWithoutExtension(file);
154 string s = run_cmd("TakeImage.py",fil);
155 if (s.Contains("OK"))
156 {
157 string[] sts = s.Split();
158 width = int.Parse(sts[0]);
159 height = int.Parse(sts[2]);
160 depth = int.Parse(sts[4]);
161 bits = int.Parse(sts[6]);
162 channels = int.Parse(sts[8]);
163
164 PixelFormat format = PixelFormat.Format8bppIndexed;
165 if (depth > 8 && bits == 2)
166 format = PixelFormat.Format16bppGrayScale;
167 else if(depth > 8 && bits == 6)
168 format = PixelFormat.Format48bppRgb;
169 else if (depth == 8 && bits == 3)
170 format = PixelFormat.Format24bppRgb;
171 else if (depth == 8 && bits == 4)
172 format = PixelFormat.Format32bppArgb;
173 bm = new AForge.Bitmap[channels];
174 for (int i = 0; i < channels; i++)
175 {
176 //Each channel is saved as a seperate file
177 bm[i] = new AForge.Bitmap(width,height,format,File.ReadAllBytes(Application.StartupPath + "/PycroManager/" + fil + i),new ZCT(), "");
178 //We delete the temporary file.
179 File.Delete(file + i);
180 }
181 return true;
182 }
183 else
184 {
185 width = 0;height = 0; depth = 0;bits = 0;channels = 0;bm = null;
186 return false;
187 }
