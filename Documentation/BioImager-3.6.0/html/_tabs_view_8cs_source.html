852 private void TabsView_KeyDown(object sender, KeyEventArgs e)
853 {
854 if (Viewer == null)
855 return;
856 double moveAmount = 1500 * (1 / App.viewer.Scale.Width);
857 if (e.KeyCode == Keys.C && e.Control)
858 {
859 Viewer.CopySelection();
860 return;
861 }
862 if (e.KeyCode == Keys.V && e.Control)
863 {
864 Viewer.PasteSelection();
865 return;
866 }
867 if (e.KeyCode == Keys.Subtract || e.KeyCode == Keys.NumPad7)
868 {
869 App.viewer.Scale = new AForge.SizeF(App.viewer.Scale.Width - 0.1f, App.viewer.Scale.Height - 0.1f);
870 }
871 if (e.KeyCode == Keys.Add || e.KeyCode == Keys.NumPad9)
872 {
873 App.viewer.Scale = new AForge.SizeF(App.viewer.Scale.Width - 0.1f, App.viewer.Scale.Height - 0.1f);
874 }
875 if (e.KeyCode == Keys.W || e.KeyCode == Keys.NumPad8)
876 {
877 Viewer.Origin = new PointD(Viewer.Origin.X, Viewer.Origin.Y - moveAmount);
878 }
879 if (e.KeyCode == Keys.S || e.KeyCode == Keys.NumPad2)
880 {
881 Viewer.Origin = new PointD(Viewer.Origin.X, Viewer.Origin.Y + moveAmount);
882 }
883 if (e.KeyCode == Keys.A || e.KeyCode == Keys.NumPad4)
884 {
885 Viewer.Origin = new PointD(Viewer.Origin.X - moveAmount, Viewer.Origin.Y);
886 }
887 if (e.KeyCode == Keys.D || e.KeyCode == Keys.NumPad6)
888 {
889 Viewer.Origin = new PointD(Viewer.Origin.X + moveAmount, Viewer.Origin.Y);
890 }
891 if (e.KeyCode == Keys.Up)
892 Microscope.MoveUp(Microscope.viewSize.Y / 2);
893 if (e.KeyCode == Keys.Down)
894 Microscope.MoveDown(Microscope.viewSize.Y / 2);
895 if (e.KeyCode == Keys.Left)
896 Microscope.MoveRight(-(Microscope.viewSize.X / 2));
897 if (e.KeyCode == Keys.Right)
898 Microscope.MoveRight((Microscope.viewSize.X / 2) / 2);
899 Viewer.UpdateStatus();
900 Viewer.UpdateView();
901 }
