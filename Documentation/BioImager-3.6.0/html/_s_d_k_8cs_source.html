355 {
356 int ret;
357
358 StringBuilder tx = new StringBuilder();
359 StringBuilder rx = new StringBuilder();
360
361 tx.Append(usertx);
362 ret = PriorScientificSDK_cmd(session,tx,rx);
363
364 if (ret == Prior.PRIOR_OK)
365 {
366 userrx = rx.ToString();
367 }
368 else
369 {
370 if (displayError)
