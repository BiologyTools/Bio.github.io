34def _get_table_value(table: bytes, key: bytes) -> bytes:
35 """Get the value for a key in a table/multiline output.
36
37 Some commands return something like a table of key/values. There
38 may be even empty lines on this table. This searches for the
39 first line with a specific key (hopefully there's only one line with such key) and returns the associated value.
40 """
41 # Key might be the first line, hence '(?:^|\r\n)'
42 match = re.search(b"(?:^|\r\n) *" + key + b": (.*)\r\n", table)
43 if match is None:
44 raise microscope.DeviceError(
45 "failed to find key %s on table: %s" % (key, table)
46 )
47 return match.group(1)

50class _iBeamConnection:
51 """Connection to a specific Toptica iBeam smart laser.
52
53 This class wraps the serial connection to the device, and provides
54 access to some of its commands performing most of the parsing and
55 validation.
56
57 Args:
58 port: port name (Windows) or path to port (everything else) to
59 connect to. For example, `/dev/ttyS1`, `COM1`, or
60 `/dev/cuad1`.
61
62 """
63
64 def __init__(self, port: str):
65 # From the Toptica iBeam SMART manual:
66 # Direct connection via COMx with 115200,8,N,1 and serial
67 # interface handshake "none". That means that no hardware
68 # handshake (DTR, RTS) and no software handshake (XON,XOFF) of
69 # the underlying operating system is supported.
70 serial_conn = serial.Serial(
71 port=port,
72 baudrate=115200,
73 timeout=1.0,
74 bytesize=serial.EIGHTBITS,
75 stopbits=serial.STOPBITS_ONE,
76 parity=serial.PARITY_NONE,
77 xonxoff=False,
78 rtscts=False,
79 dsrdtr=False,
80 )
81 self._serial = microscope._utils.SharedSerial(serial_conn)
82
83 # We don't know what is the current verbosity state and so we
84 # don't know yet what we should be reading back. So blindly
85 # set to the level we want, flush all output, and then check
86 # if indeed this is a Toptica iBeam device.
87 with self._serial.lock:
88 self._serial.write(b"echo off\r\n")
89 self._serial.write(b"prompt off\r\n")
90 # The talk level we want is 'usual'. In theory we should
91 # be able to use 'quiet' which only answers queries but in
92 # practice 'quiet' does not answer some queries like 'show
93 # serial'.
94 self._serial.write(b"talk usual\r\n")
95 self._serial.readlines() # discard all pending lines
96
97 # Empty command does nothing and returns nothing extra so we
98 # use it to ensure this at least behaves like a Toptica iBeam.
99 try:
100 self.command(b"")
101 except microscope.DeviceError as e:
102 raise microscope.InitialiseError(
103 "Failed to confirm Toptica iBeam on %s" % (port)
104 ) from e
105
106 answer = self.command(b"show serial")
107 if not answer.startswith(b"SN: "):
108 raise microscope.DeviceError(
109 "Failed to parse serial from %s" % answer
110 )
111 _logger.info("got connection to Toptica iBeam %s", answer.decode())
112
113 def command(self, command: bytes) -> bytes:
114 """Run command and return answer after minimal validation.
115
116 The output of a command has the format::
117
118 \r\nANSWER\r\n[OK]\r\n
119
120 The returned bytes only include `ANSWER` without its own final
121 `\r\n`. This means that the return value might be an empty
122 array of bytes.
123 """
124 # We expect to be on 'talk usual' mode without prompt so each
125 # command will end with [OK] on its own line.
126 with self._serial.lock:
127 self._serial.write(command + b"\r\n")
128 # An answer always starts with \r\n so there will be one
129 # before [OK] even if this command is not a query.
130 answer = self._serial.read_until(b"\r\n[OK]\r\n")
131
132 if not answer.startswith(b"\r\n"):
133 raise microscope.DeviceError(
134 "answer to command %s does not start with CRLF."
135 " This may be leftovers from a previous command:"
136 " %s" % (command, answer)
137 )
138 if not answer.endswith(b"\r\n[OK]\r\n"):
139 raise microscope.DeviceError(
140 "Command %s failed or failed to read answer: %s"
141 % (command, answer)
142 )
143
144 # If an error occurred, the answer still ends in [OK]. We
145 # need to check if the second line (first line is \r\n) is an
146 # error code with the format "%SYS-L-XXX, error description"
147 # where L is the error level (I for Information, W for
148 # Warning, E for Error, and F for Fatal), and XXX is the error
149 # code number.
150 if answer[2:7] == b"%SYS-" and answer[7] != ord(b"I"):
151 # Errors of level I (information) should not raise an
152 # exception since they can be replies to normal commands.
153 raise microscope.DeviceError(
154 "Command %s failed: %s" % (command, answer)
155 )
156
157 # Exclude the first \r\n, the \r\n from a possible answer, and
158 # the final [OK]\r\n
159 return answer[2:-8]

170 """Set power in mW for channel 2 (normal operating level channel).

178 def show_power_uW(self) -> float:
179 """Returns actual laser power in µW."""
180 answer = self.command(b"show power")
181 if not answer.startswith(b"PIC = ") and not answer.endswith(b" uW "):
182 raise microscope.DeviceError(
183 "failed to parse power from answer: %s" % answer
184 )
185 return float(answer[7:-5])

191 def show_max_power(self) -> float:
192 # There should be a cleaner way to get these, right? We can
193 # query the current limits (mA) but how do we go from there to
194 # the power limits (mW)?
195 table = self.command(b"show satellite")
196 value = _get_table_value(table, key=b"Pmax")
197 if not value.endswith(b" mW"):
198 raise microscope.DeviceError(
199 "failed to parse power from %s" % value
200 )
201 return float(value[:-3])

204class TopticaiBeam(
205 microscope._utils.OnlyTriggersBulbOnSoftwareMixin,
206 microscope.abc.LightSource,
207):
208 """Toptica iBeam smart laser.
209
210 Control of laser power is performed by setting the power level on
211 the normal channel (#2) only. The bias channel (#1) is left
212 unmodified and so defines the lowest level power.
213 """
214
215 def __init__(self, port: str, **kwargs) -> None:
216 super().__init__(**kwargs)
217 self._conn = _iBeamConnection(port)
218 # The Toptica iBeam has up to five operation modes, named
219 # "channels" on the documentation. Only the first three
220 # channels have any sort of documentation:
221 #
222 # Ch 1: bias level channel
223 # Ch 2: normal operating level channel
224 # Ch 3: only used at high-power models
225 #
226 # We haven't come across a laser with a channel 3 so we are
227 # ignoring it until then. So we just leave the bias channel
228 # (1) alone and control power via the normal channel (2).
229 self._max_power = self._conn.show_max_power()

232 pass
