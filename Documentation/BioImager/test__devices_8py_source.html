20"""Test all the concrete device classes.
21
22We have the same tests for all devices of the same type. To do this,
23there is a :class:`unittest.TestCase` class for each device thatsubclasses from that device type class of tests. Each such class only
24needs to implement the `setUp` method. It may also add device
25specific tests.
26
27Using lasers as example, there is a :class:`.LightSourceTests` class
28full of `test_*` methods, each of them a test on its own. For each
29light source device supported there is one test class, e.g.,
30`TestOmicronDeepstarLaser`, and `TestCoherentSapphireLaser`. These
31subclass from both :class:`unittest.TestCase` and `LightSourceTests`
32and need only to implement `setUp` which sets up the fake and
33constructs the device instance required to run the tests.
34
35"""

47class TestSerialMock(unittest.TestCase):
48 # Our tests for serial devices depend on our SerialMock base
49 # class working properly so yeah, we need tests for that too.
50 class Serial(mocks.SerialMock):
51 eol = b"\r\n"

53 def handle(self, command):
54 if command.startswith(b"echo "):
55 self.in_buffer.write(command[5:] + self.eol)
56 elif command in [b"foo", b"bar"]:
57 pass
58 else:
59 raise RuntimeError("unknown command '%s'" % command.decode())
60
61 def setUp(self):
62 self.serial = TestSerialMock.Serial()
63 patcher = unittest.mock.patch.object(
64 TestSerialMock.Serial, "handle", wraps=self.serial.handle
65 )
66 self.addCleanup(patcher.stop)
67 self.mock = patcher.start()
68

72

75 self.serial.write(b"o")

158class LightSourceTests(DeviceTests):
159 """Base class for :class:`LightSource` tests.
160
161 This class implements all the general laser tests and is meant to
162 be mixed with :class:`unittest.TestCase`. Subclasses must
163 implement the `setUp` method which must add two properties:
164
165 `device`
166 Instance of the :class:`LightSource` implementation being
167 tested.
168
169 `fake`
170 Object with a multiple attributes that specify the hardware
171 and control the tests, such as the device max and min power
172 values. Such attributes may as well be attributes in the
173 class that fakes the hardware.
174
175 """
176
177 def assertEqualMW(self, first, second, msg=None):
178 # We could be smarter, but rounding the values should be
179 # enough to check the values when comparing power levels.
180 self.assertEqual(round(first), round(second), msg)
181
182 def test_get_is_on(self):
183 self.assertEqual(self.device.connection.light, self.device.get_is_on())
184 self.device.enable()
185 self.assertEqual(self.device.connection.light, self.device.get_is_on())
186 self.device.disable()
187 self.assertEqual(self.device.connection.light, self.device.get_is_on())
188
189 def test_off_after_constructor(self):
190 # Some lasers, such as our Coherent Sapphire emit laser
191 # radiation as soon as the key is switched on. We should
192 # ensure that the laser is turned off during the
193 # construction.
194 self.assertFalse(self.device.get_is_on())
195
196 def test_turning_on_and_off(self):
197 self.device.enable()
198 self.assertTrue(self.device.get_is_on())
199 self.device.disable()
200 self.assertFalse(self.device.get_is_on())
201
202 def test_shutdown(self):
203 self.device.enable()
204 self.device.disable()
205 self.device.shutdown()
206
207 def test_power_when_off(self):
208 self.device.disable()
209 self.assertIsInstance(self.device.power, float)
210 self.assertEqual(self.device.power, 0.0)
211
212 def test_setting_power(self):
213 self.device.enable()
214 self.assertIsInstance(self.device.power, float)
215 power_mw = self.device.power * self.fake.max_power
216 self.assertEqualMW(power_mw, self.fake.default_power)
217 self.assertEqualMW(self.device.power, self.device.get_set_power())
218
219 new_power = 0.5
220 new_power_mw = new_power * self.fake.max_power
221 self.device.power = new_power
222 self.assertEqualMW(
223 self.device.power * self.fake.max_power, new_power_mw
224 )
225 self.assertEqualMW(new_power, self.device.get_set_power())
226
227 def test_setting_power_outside_limit(self):
228 self.device.enable()
229 self.device.power = -0.1
230 self.assertEqual(
231 self.device.power,
232 self.fake.min_power / self.fake.max_power,
233 "clip setting power below 0",
234 )
235 self.device.power = 1.1
236 self.assertEqual(self.device.power, 1.0, "clip setting power above 1")
237
238 def test_status(self):
239 status = self.device.get_status()
240 self.assertIsInstance(status, list)
241 for msg in status:
242 self.assertIsInstance(msg, str)

269class DeformableMirrorTests(DeviceTests):
270 """Collection of test cases for deformable mirrors.
271
272 Should have the following properties defined during `setUp`:
273 `planned_n_actuators` (int): number of actuators
274 `device` (DeformableMirror): the microscope device instance
275 `fake`: an object with the method `get_current_pattern`
276 """
277
278 def assertCurrentPattern(self, expected_pattern, msg=""):
279 numpy.testing.assert_array_equal(
280 self.fake.get_current_pattern(), expected_pattern, msg
281 )
282
283 def test_get_number_of_actuators(self):
284 self.assertIsInstance(self.device.n_actuators, int)
285 self.assertGreater(self.device.n_actuators, 0)
286 self.assertEqual(self.device.n_actuators, self.planned_n_actuators)
287
288 def test_applying_pattern(self):
289 pattern = numpy.full((self.planned_n_actuators,), 0.2)
290 self.device.apply_pattern(pattern)
291 self.assertCurrentPattern(pattern)
292
293 def test_out_of_range_pattern(self):
294 # While we expect values in the [0 1] range, we should not
295 # actually be checking for that.
296 pattern = numpy.zeros((self.planned_n_actuators,))
297 for v in [-1000, -1, 0, 1, 3]:
298 pattern[:] = v
299 self.device.apply_pattern(pattern)
300 self.assertCurrentPattern(pattern)
301
302 def test_software_triggering(self):
303 n_patterns = 5
304 patterns = numpy.random.rand(n_patterns, self.planned_n_actuators)
305 self.device.queue_patterns(patterns)
306 for i in range(n_patterns):
307 self.device.next_pattern()
308 self.assertCurrentPattern(patterns[i])
309
310 def test_validate_pattern_too_long(self):
311 patterns = numpy.zeros((self.planned_n_actuators + 1))
312 with self.assertRaisesRegex(Exception, "length of second dimension"):
313 self.device.apply_pattern(patterns)
314
315 def test_validate_pattern_swapped_dimensions(self):
316 patterns = numpy.zeros((self.planned_n_actuators, 1))
317 with self.assertRaisesRegex(Exception, "length of second dimension"):
318 self.device.apply_pattern(patterns)
319
320 def test_validate_pattern_with_extra_dimension(self):
321 patterns = numpy.zeros((2, 1, self.planned_n_actuators))
322 with self.assertRaisesRegex(
323 Exception, "dimensions \\(must be 1 or 2\\)"
324 ):
325 self.device.apply_pattern(patterns)
