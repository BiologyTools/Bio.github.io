153 /* It loads a .NET assembly and provides a dictionary of all the types, modules, constructors,
154 interfaces, enums, properties, fields, methods, and members. */
155 public Lib(string file)
156 {
157 dll = Assembly.LoadFile(file);
158 Type[] tps = dll.GetExportedTypes();
159 foreach (Type type in tps)
160 {
161 TypeInfo t = new TypeInfo();
162 t.type = type;
163 string s = type.ToString();
164 s = s.Remove(0, s.LastIndexOf('.') + 1);
165 if (type.IsEnum)
166 {
167 GetEnums(t);
168 }
169 GetInterfaces(t);
170 GetConstructors(t);
171 GetProperties(t);
172 GetFields(t);
173 GetMethods(t);
174 GetMembers(t);
175 Types.Add(s, t);
176 }
177 Module[] mds = dll.GetModules();
178 foreach (Module m in mds)
179 {
180 string s = m.ToString();
181 s = s.Remove(0, s.LastIndexOf('.') + 1);
182 Modules.Add(s, m);
183 }
184 }
