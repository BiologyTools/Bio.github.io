62 public TabsView(string[] arg)
63 {
64 InitializeComponent();
65 LoadProperties();
66 Init();
67 if (arg.Length == 0)
68 return;
69 else
70 {
71 for (int i = 0; i < arg.Length; i++)
72 {
73 if (arg[i].EndsWith(".cs"))
74 {
75 App.runner.RunScriptFile(arg[0]);
76 return;
77 }
78 if (arg[i].EndsWith(".ijm"))
79 {
80 ImageJ.RunMacro(arg[i], "");
81 return;
82 }
83 else
84 {
85 BioImage.OpenFile(arg[i], true);
86 }
87 }
88 }

97 {
98 if (b == null)
99 return;
100 if (b.filename.Contains("/") || b.filename.Contains("\\"))

102 TabPage t = new TabPage(b.filename);
103 ImageView v = new ImageView(b);
104 v.ShowStage = false;
105 v.Dock = DockStyle.Fill;

115 }

186 /// If the image is pyramidal, add 42 to the width and 206 to the height. Otherwise, add 20 to
187 /// the width and 180 to the height
188 ///
189 /// @return The size of the image.
190 public void ResizeView()
191 {
192 if (Image == null)
193 return;
194 System.Drawing.Size s;
195 if (SelectedImage.isPyramidal)
196 s = new System.Drawing.Size(ImageView.SelectedImage.Resolutions[Viewer.Level].SizeX + 42, ImageView.SelectedImage.Resolutions[Viewer.Level].SizeY + 206);
197 else
198 s = new System.Drawing.Size(ImageView.SelectedImage.SizeX + 20, ImageView.SelectedImage.SizeY + 180);
199 if (s.Width > Screen.PrimaryScreen.Bounds.Width || s.Height > Screen.PrimaryScreen.Bounds.Height)
200 {
201 this.WindowState = FormWindowState.Maximized;
202 Viewer.GoToImage();
203 }
204 else
205 {
206 Size = s;
207 Viewer.GoToImage();
208 }

900 /// If the user presses a key, and the key is one of the arrow keys, then move the microscope in
901 /// the direction of the arrow key
902 ///
903 /// @param sender The object that raised the event.
904 /// @param KeyEventArgs e
905 ///
906 /// @return The return value is a string.
907 private void TabsView_KeyDown(object sender, KeyEventArgs e)
908 {
909 if (Viewer == null)
910 return;
911 double moveAmount = 1500 * (1 / App.viewer.Scale.Width);
912 if (e.KeyCode == Keys.C && e.Control)
913 {
914 Viewer.CopySelection();
915 return;
916 }
917 if (e.KeyCode == Keys.V && e.Control)
918 {
919 Viewer.PasteSelection();
920 return;
921 }
922 if (e.KeyCode == Keys.Subtract || e.KeyCode == Keys.NumPad7)
923 {
924 App.viewer.Scale = new AForge.SizeF(App.viewer.Scale.Width - 0.1f, App.viewer.Scale.Height - 0.1f);
925 }
926 if (e.KeyCode == Keys.Add || e.KeyCode == Keys.NumPad9)
927 {
928 App.viewer.Scale = new AForge.SizeF(App.viewer.Scale.Width - 0.1f, App.viewer.Scale.Height - 0.1f);
929 }
930 if (e.KeyCode == Keys.W || e.KeyCode == Keys.NumPad8)
931 {
932 Viewer.Origin = new PointD(Viewer.Origin.X, Viewer.Origin.Y - moveAmount);
933 }
934 if (e.KeyCode == Keys.S || e.KeyCode == Keys.NumPad2)
935 {
936 Viewer.Origin = new PointD(Viewer.Origin.X, Viewer.Origin.Y + moveAmount);
937 }
938 if (e.KeyCode == Keys.A || e.KeyCode == Keys.NumPad4)
939 {
940 Viewer.Origin = new PointD(Viewer.Origin.X - moveAmount, Viewer.Origin.Y);
941 }
942 if (e.KeyCode == Keys.D || e.KeyCode == Keys.NumPad6)
943 {
944 Viewer.Origin = new PointD(Viewer.Origin.X + moveAmount, Viewer.Origin.Y);
945 }
946 if (e.KeyCode == Keys.Up)
947 Microscope.MoveUp(Microscope.viewSize.Y / 2);
948 if (e.KeyCode == Keys.Down)
949 Microscope.MoveDown(Microscope.viewSize.Y / 2);
950 if (e.KeyCode == Keys.Left)
951 Microscope.MoveRight(-(Microscope.viewSize.X / 2));
952 if (e.KeyCode == Keys.Right)
953 Microscope.MoveRight((Microscope.viewSize.X / 2) / 2);
954 Viewer.UpdateStatus();
955 Viewer.UpdateView();
956 }
