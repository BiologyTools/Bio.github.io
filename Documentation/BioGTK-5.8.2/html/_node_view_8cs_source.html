127 public void UpdateItems()
128 {
129 Gtk.TreeStore store = new Gtk.TreeStore(typeof(string), typeof(string), typeof(string));
130
131 foreach (BioImage b in Images.images)
132 {
133 Gtk.TreeIter iter = store.AppendValues(System.IO.Path.GetFileName(b.Filename));
134 foreach (AForge.Bitmap bit in b.Buffers)
135 {
136 if(bit.Plane != null)
137 store.AppendValues(iter, bit.Coordinate.ToString(), bit.Plane.Exposure, bit.Plane.Delta);
138 else
139 store.AppendValues(iter, bit.Coordinate.ToString(), 0, 0);
140 }
